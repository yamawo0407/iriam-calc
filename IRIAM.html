            <label>残り日数（6〜0）</label>

            <input type="number" id="days" value="6" min="0" max="6">

        </div>

        <div class="input-group">

            <label>現在の累計スコア</label>

            <input type="number" id="score" value="0">

        </div>

        <button onclick="calculate()">シミュレーション実行</button>


        <div class="results">

            <table id="res-table">

                <thead><tr><th>日</th><th>ランク</th><th>残り</th><th>累計</th><th>判定</th></tr></thead>

                <tbody></tbody>

            </table>

    <script>

        const RANKS = ["D", "C1", "C2", "C3", "C4", "C5", "B1", "B2", "B3", "A1", "A2", "A3", "S1", "S2", "S3"];

        function calculate() {

            let curRank = document.getElementById('rank').value;

            let dayRem = parseInt(document.getElementById('days').value);

            let total = parseInt(document.getElementById('score').value);

            const tbody = document.querySelector("#res-table tbody");

            tbody.innerHTML = "";


            for (let i = 1; i <= 14; i++) {

                let status = "";

                let displayDay = dayRem;

                let isEnd = false;


                // 簡易シミュレーションのため1日2pt獲得と仮定

                total += 2; 


                if (total >= 18) {

                    status = "アップ";

                    isEnd = true;

                } else if (dayRem <= 0) {

                    status = (total >= 12) ? "キープ" : "ダウン";

                    isEnd = true;

                } else {

                    dayRem--;

                }


                let tr = `<tr><td>${i}日後</td><td>${curRank}</td><td>残り${displayDay}日</td><td>${total}</td><td class="${status === 'アップ' ? 'up' : (status === 'ダウン' ? 'down' : '')}">${status}</td></tr>`;

                tbody.innerHTML += tr;


                if (isEnd) {

                    let idx = RANKS.indexOf(curRank);

                    if (status === "アップ" && idx < RANKS.length - 1) curRank = RANKS[idx + 1];

                    else if (status === "ダウン" && idx > 0) curRank = RANKS[idx - 1];

                    total = 0;

                    dayRem = 6;

                }

            }

        }

    </script>
